public boolean funct(int n, int l, int u, double x[], double f[])

{

int i;

double x1,x2,x3;



x1 = (l == 1) ? 0.0 : x[l-1];

x2=x[l];

x3 = (l == n) ? 0.0 : x[l+1];

for (i=l; i<=u; i++) {

f[i]=(3.0-2.0*x2)*x2+1.0-x1-x3*2.0;

x1=x2;

x2=x3;

x3 = (i <= n-2) ? x[i+2] : 0.0;

}

return (true);

}
