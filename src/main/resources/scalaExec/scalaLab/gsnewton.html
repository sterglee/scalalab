public boolean funct(int m, int n, double par[], double g[])

{

int i;



for (i=1; i<=m; i++) {

if (par[3]*x[i] > 680.0) return false;

g[i]=par[1]+par[2]*Math.exp(par[3]*x[i])-y[i];

}

return true;

}

public void jacobian(int m, int n, double par[],

double g[], double jac[][])

{

int i;

double ex;



for (i=1; i<=m; i++) {

jac[i][1]=1.0;

jac[i][2]=ex=Math.exp(par[3]*x[i]);

jac[i][3]=x[i]*par[2]*ex;

}

}
